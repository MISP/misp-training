
\begin{frame}
    \frametitle{What can I do on the instance?}

    The MN MISP instance is there to \textbf{support your investigations} and to
    \textbf{connect you with the rest of the community}.

    \begin{center}
        \begin{columns}[T]
            \begin{column}{0.5\textwidth}
                \card{\linewidth}{1.5em}{\faSearch \, \textbf{Search}}
                    {
                        {\setlength{\leftmargini}{1.0em}
                        \begin{itemize}
                            \item Look up \textbf{indicators}
                            \item Query \textbf{knowledge-base} items relevant to your incidents
                        \end{itemize}
                        }
                    }
            \end{column}
            \hfill
            \begin{column}{0.5\textwidth}
                \card{\linewidth}{1.5em}{\faProjectDiagram \, \textbf{Correlate}}
                    {
                        {\setlength{\leftmargini}{1.0em}
                        \begin{itemize}
                            \item Find \textbf{related incidents}
                            \item See who else is affected
                        \end{itemize}
                        }
                    }
            \end{column}
        \end{columns}
        \begin{columns}[T]
            \begin{column}{0.5\textwidth}
                \card{\linewidth}{1.5em}{\faCogs \, \textbf{Operationalise}}
                    {
                        {\setlength{\leftmargini}{1.0em}
                        \begin{itemize}
                            \item Build \textbf{remediation} \& \textbf{prevention} rules
                            \item Feed data-sets to your tools
                        \end{itemize}
                        }
                    }
            \end{column}
            \hfill
            \begin{column}{0.5\textwidth}
                \card{\linewidth}{1.5em}{\faComments \, \textbf{Collaborate}}
                    {
                        {\setlength{\leftmargini}{1.0em}
                        \begin{itemize}
                            \item Open \textbf{dialogues} with other members
                            \item Ask, support, contribute
                        \end{itemize}
                        }
                    }
            \end{column}
        \end{columns}
    \end{center}
\end{frame}


\begin{frame}
    \frametitle{Interacting with the community}

    Every event is an opportunity to \textbf{open a dialogue} with the members
    who produced or are affected by it.

    \vspace{0.5em}

    \begin{center}
        \begin{columns}[T]
            \begin{column}{0.49\textwidth}
                \card{\linewidth}{2.5em}
                    {{\Large\faQuestionCircle}\hspace{0.3em}\textbf{Ask for more}}
                    {
                        \begin{itemize}
                            \item Request additional information
                            \item Ask for support on an incident
                        \end{itemize}
                    }
            \end{column}
            \begin{column}{0.49\textwidth}
                \card{\linewidth}{2.5em}
                    {{\Large\faLightbulb}\hspace{0.3em}\textbf{Give back}}
                    {
                        \begin{itemize}
                            \item Provide improvements or hints
                            \item Correct or enrich existing data
                        \end{itemize}
                    }
            \end{column}
        \end{columns}

        \vspace{-0.5em}

        \begin{columns}[T]
            \begin{column}{0.7\textwidth}
                \card{\linewidth}{2.5em}
                    {{\Large\faBullhorn}\hspace{0.3em}\textbf{Signal activity}}
                    {
                        \begin{itemize}
                            \item Simply let the community know that a threat
                                  is \textbf{still active in the wild}
                        \end{itemize}
                    }
            \end{column}
        \end{columns}
    \end{center}

    \vspace{0.8em}

    \centering
    \small
    A single sighting can tell dozens of teams that a threat is not yet over.
\end{frame}
